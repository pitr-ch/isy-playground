def link_to(label, &action)
  if label.kind_of?(String)
    a label, :href => "#", :data-action-id' => register_action(&action)
  elsif label.kind_of?(Proc)
    a :href => "#", :data-action-id' => register_action(&action), &label
  else
    raise ArgumentError
  end
end
