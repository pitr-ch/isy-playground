def switch_packed
  unless @packed = !@packed
    @variables = inspector \
        ::Hash[obj.instance_variables.map {|name| [name, obj.instance_variable_get(name)]}.flatten(1)],
        'Instance variables'
  else
    @variables = nil
  end
end
