def content_with_wrapper
  if self.class.wrapped_in
    send self.class.wrapped_in, wrapper_options do
      content
    end
  else
    content
  end
end

def render(obj)
  widget begin
    if obj.kind_of?(Erector::Widget)
      obj
    elsif obj.respond_to?(:widget)
      obj.widget
    else
      raise ArgumentError, obj.inspect
    end
  end
end
